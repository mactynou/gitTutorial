git - команда для вызова информации о всех командах
git add -A - подготовка к отправлению файла в git
git commit -m - вроде сохранение изменений
git status - статус об изменении отправляемых файлов
git push - команда для отправки файлов в git

echo >> имя_файла - создание файла
mkdir имя_файла - создание каталога
del имя_файла - удаление файла
rmdir имя_файла - удаление каталога
